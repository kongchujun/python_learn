<clickhouse>
    <profiles>
        <!-- 
            这里的 'default' 是指默认的用户配置档案。
            当你用 clickhouse-client 连接而未指定用户时，就使用这个profile。
        -->
        <default>
            <!-- 
                将这个标志设置为1，表示允许执行实验性的磁盘管理命令。
                这就是解决 "NOT_IMPLEMENTED" 错误的关键。
            -->
            <allow_experimental_disk_management_commands>1</allow_experimental_disk_management_commands>
        </default>
    </profiles>
</clickhouse>
