<template>
    <div class="company-form">
      <div class="inputs">
        <input type="text" v-model="companyId" placeholder="Company ID" class="input-field" />
        <button @click="submitCompanyQuery" class="submit-button">Submit</button>
      </div>
      <table v-if="results.length > 0" class="results-table">
        <thead>
          <tr>
            <th>Company Name</th>
            <th>PermID</th>
            <th>Details</th>
          </tr>
        </thead>
        <tbody>
          <tr v-for="(result, index) in results" :key="index">
            <td>{{ result.companyName }}</td>
            <td>{{ result.permId }}</td>
            <td><a :href="result.link" target="_blank" class="link">{{ result.link }}</a></td>
          </tr>
        </tbody>
      </table>
    </div>
  </template>
  
  <script setup>
  import { ref } from 'vue';
  import axios from 'axios';
  
  const companyId = ref('');
  const results = ref([]);
  
  const submitCompanyQuery = () => {
    const params = {
      companyId: companyId.value
    };
  
    axios.get('http://localhost:8123/permid', { params })
      .then(response => {
        results.value = response.data;
      })
      .catch(error => {
        console.error('There was an error fetching the data:', error);
      });
  };
  </script>
