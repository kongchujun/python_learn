import psycopg2

# 数据库连接参数
conn_params = {
    "dbname": "your_database_name",
    "user": "your_username",
    "password": "your_password",
    "host": "localhost",  # 或者其他数据库服务器地址
    "port": "5432"         # 根据实际情况调整端口
}

# 创建连接
conn = psycopg2.connect(**conn_params)
cursor = conn.cursor()

# 查询语句，获取唯一的公司名称
query_sql = "SELECT DISTINCT company FROM company_info;"

# 执行查询操作
cursor.execute(query_sql)

# 获取所有唯一的公司名称
unique_companies = cursor.fetchall()

# 输出结果
for company in unique_companies:
    print(company[0])

# 关闭游标和连接
cursor.close()
conn.close()
