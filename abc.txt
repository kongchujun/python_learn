from sqlalchemy import create_engine, text

# 创建数据库连接引擎
engine = create_engine('postgresql+psycopg2://your_username:your_password@your_host:your_port/your_dbname')

# 删除符合条件的数据
with engine.connect() as connection:
    delete_query = text("DELETE FROM your_table_name WHERE source = :source")
    connection.execute(delete_query, {"source": 'a'})
