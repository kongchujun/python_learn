import psycopg2

# 连接到PostgreSQL数据库
conn = psycopg2.connect(
    dbname='your_dbname',
    user='your_username',
    password='your_password',
    host='your_host',
    port='your_port'
)

# 创建游标对象
cur = conn.cursor()

# 执行DELETE语句删除符合条件的行
cur.execute("""
DELETE FROM your_table_name
WHERE source = 'a'
""")

# 提交事务
conn.commit()

# 关闭游标和连接
cur.close()
conn.close()
